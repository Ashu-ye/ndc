###TO block ping/ICMP 

iptables -D FORWARD 1
iptables -A FORWARD -j REJECT
iptables -L FORWARD
iptables -D INPUT 2   # To block ICMP

###TO allow ICMP
iptables -I FORWARD 3 -p icmp -s 192.168.25.0/24 -d 192.168.72.20 -j ACCEPT
iptables -I FORWARD 3 -p icmp -d 192.168.25.0/24 -s 192.168.72.20 -j ACCEPT


###Adding rules to come traffic from outside

iptables -I FORWARD 1 -p udp -s 192.168.25.0/24 -d 192.168.72.20 --dport=53 -j ACCEPT 
iptables -I FORWARD 2 -p udp -d 192.168.25.0/24 -s 192.168.72.20 --dport=53 -j ACCEPT
iptables -L FORWARD -n -v
